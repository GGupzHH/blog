    GetPage(
      name: AppRoutes.Login, 
      page: () => const LoginView(), 
      // 定义转场动画
      transition: Transition.downToUp

  ];

  static final unknownRoute = GetPage(name: AppRoutes.Notfound, page: () => const NotFoundView());
}
